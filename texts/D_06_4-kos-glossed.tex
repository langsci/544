\ea\label{ex:nreximeldan}
Imelda numu rileyo nake. Numu kweyo eku nake. Kos sueli, kos papi.\\
\gll Imelda	numu	rileyo	nake	numu	kwe-yo	eku\_nake	kos	sueli	kos	papi\\
     Imelda	oil.palm	above	sit	oil.palm	palm.rib-\textsc{loc}	sit.down	broom	cut	broom	make.\textsc{pl}.\textsc{o}\\
\glt `Imelda is sitting above in a \textit{numu}-palm. She is sitting on a palm rib to cut a broom, to make a broom.'
