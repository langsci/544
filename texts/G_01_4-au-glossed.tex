\ea\label{ex:nrexgreeny}
Ko loro Greenyo. Ol ba ikoiele! Ko ermue riyo. Ko ermue pulo Greenyo.\\
\gll ko	lo=ro	Green-yo	ol	ba	ikoiele	ko	ermue	riyo	ko	ermue	pulo	Green-yo\\
     I	go.\textsc{pp}=\textsc{emph}	Green-\textsc{loc}	mountain	other	very.big	I	for.the.first.time	see.\textsc{o}.\textsc{inanim}.\textsc{pp}	I	for.the.first.time	come.\textsc{pp}	Green-\textsc{loc}\\
\glt `I went to Green. Such big mountains! I saw them for the first time. I came to Green for the first time.'
\z

\ea\label{ex:nrexruelee}
Kukûnapno dob riyo pu po. Ru ikoiele po. Ol sowelayo.\\
\gll k-kûne-p-no	dob	riyo	pu	po	ru	ikoiele	po	ol	sowelayo\\
     \textsc{sub}-go.down-\textsc{pc}-\textsc{co}	eye	see.\textsc{o}.\textsc{inanim}.\textsc{pp}	rain	\textsc{lv}.\textsc{pp}	fog	very.big	\textsc{lv}.\textsc{pp}	mountain	cover.thoroughly.\textsc{pp}\\
\glt `When (the plane) had gone down I saw (that) it rained. (There was) dense fog. It covered up the mountains.'
\z
